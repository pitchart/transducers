Imperative programming

$collection = [1, [2], [3, 4]];

function flatten(iterable $collection) {
    $results = [];
    foreach ($collecion as $item) {
        if ($item === null) {
            continue;
        }
        return array_merge($results, is_iterable($item) ? flatten($item) : [$item]);
    }
    return $results;
};

function process(iterable $collection) {
    $results = [];
    foreach (flatten($collection) as $item) {
        $result = $item + 1;
        if ($result > 3) {
            $results[] = $result;
        }
    }
}

process([1, [2], [3, 4]]); // [4, 5]
// WARNING: cyclomatic complexity
