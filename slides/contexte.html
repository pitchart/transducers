Imperative programming

$cities = [];
foreach ($users as $user) {
    if ($user->getAddresses() !== null) {
        foreach ($user->getAddresses() as $address) {
            if ($address->isActive()) {
                $cities[] = new City($address->getZip(), $address->getCity());
            }
        }
    }
} // WARNING: cyclomatic complexity

VS Declarative programming

SELECT zip, city
FROM address
INNER JOIN user
    ON address.user = user.id
WHERE address.is_active = 1
